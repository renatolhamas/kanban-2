## Connect to instance
const options = {
  method: 'POST',
  headers: {
    apikey: '4e6a8d1b-3c7f-4b2a-9e12-8d5c4b3a9f01',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    immediate: true,
    phone: '<string>',
    subscribe: ['<string>'],
    webhookUrl: '<string>'
  })
};

fetch('http://localhost:8080/instance/connect', options)
  .then(res => res.json())
  .then(res => console.log(res))
  .catch(err => console.error(err));
