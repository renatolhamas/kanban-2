## Create a new instance
const options = {
  method: 'POST',
  headers: {apikey: '<api-key>', 'Content-Type': 'application/json'},
  body: JSON.stringify({
    name: '<string>',
    proxy: {
      address: '<string>',
      password: '<string>',
      port: '<string>',
      username: '<string>'
    },
    token: '<string>'
  })
};

fetch('http://localhost:8080/instance/create', options)
  .then(res => res.json())
  .then(res => console.log(res))
  .catch(err => console.error(err));
