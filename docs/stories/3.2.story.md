# Story 3.2: Evo GO Pairing (Page Fully Functional)

## Story Statement

**As a** owner,  
**I want to** connect my WhatsApp account via a fully functional pairing page with real-time status updates,  
**so that** I can pair my account and immediately start receiving messages from Evo GO webhooks.

### API Route Patterns

#### Status Endpoint (GET)

```typescript
// app/api/settings/evo-go/status/route.ts
import { auth } from "@/lib/middleware/auth";
import { tenantIsolation } from "@/lib/middleware/tenant-isolation";
import { createClient } from "@supabase/supabase-js";

export async function GET(req: Request) {
  try {
    const user = await auth(req);
    const { tenantId } = await tenantIsolation(req, user);
    
    const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY);
    const { data, error } = await supabase
      .from("tenants")
      .select("connection_status, evolution_instance_id")
      .eq("id", tenantId)
      .single();
    
    if (error) throw error;
    
    return Response.json({
      connection_status: data.connection_status,
      evolution_instance_id: data.evolution_instance_id
    }, { status: 200 });
  } catch (error) {
    if (error instanceof AuthError) return new Response(null, { status: 401 });
    if (error instanceof TenantIsolationError) return new Response(null, { status: 403 });
    return new Response(null, { status: 500 });
  }
}
```

#### Webhook Endpoint (POST)

```typescript
// app/api/webhooks/evo-go/route.ts
import { validateEvoGoSignature } from "@/lib/api/evo-go-signature";
import { createClient } from "@supabase/supabase-js";

export async function POST(req: Request) {
  try {
    const body = await req.text();
    const signature = req.headers.get("x-signature");
    if (!signature) return new Response(null, { status: 401 });
    
    const secret = process.env.EVO_GO_WEBHOOK_SECRET!;
    if (!validateEvoGoSignature(body, signature, secret)) {
      return new Response(null, { status: 401 });
    }
    
    const payload = JSON.parse(body);
    const { event, data } = payload;
    
    const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY);
    
    // Extract tenantId from URL query parameter
    const url = new URL(req.url);
    const tenantId = url.searchParams.get("tenantId");
    if (!tenantId) return new Response("Missing tenantId", { status: 400 });

    switch (event) {
      case "CONNECTION_UPDATE":
        await supabase
          .from("tenants")
          .update({ connection_status: data.status || "disconnected" })
          .eq("id", tenantId);
        console.log(`[Webhook] CONNECTION_UPDATE: tenant=${tenantId}, status=${data.status}`);
        break;
      
      case "QRCODE_UPDATED":
        console.log(`[Webhook] QRCODE_UPDATED: ${JSON.stringify(data)}`);
        break;
      
      case "MESSAGES_UPSERT":
        console.log(`[Webhook] MESSAGES_UPSERT: (handled in Story 3.5)`);
        break;
      
      default:
        console.log(`[Webhook] Unknown event: ${event}`);
    }
    
    return new Response(null, { status: 200 });
  } catch (error) {
    console.error("[Webhook] Error:", error);
    return new Response(null, { status: 200 });
  }
}
```

### Frontend Component Pattern

```typescript
// app/settings/page.tsx
"use client";

import { useState, useEffect } from "react";
import { Card, Button, Badge, ConfirmModal } from "@/components/common";

type ConnectionState = "idle" | "loading" | "qr_displayed" | "expired" | "connected" | "error";

export default function SettingsPage() {
  const [state, setState] = useState<ConnectionState>("idle");
  const [qrCode, setQrCode] = useState<string | null>(null);
  const [timerSeconds, setTimerSeconds] = useState(300); // 5 min
  const [error, setError] = useState<string | null>(null);
  const [connectionStatus, setConnectionStatus] = useState<string>("disconnected");
  
  useEffect(() => {
    if (state !== "qr_displayed") return;
    
    const pollInterval = setInterval(async () => {
      try {
        const response = await fetch("/api/settings/evo-go/status");
        if (!response.ok) throw new Error(`Status: ${response.status}`);
        
        const { connection_status } = await response.json();
        setConnectionStatus(connection_status);
        
        if (connection_status === "connected") {
          setState("connected");
          clearInterval(pollInterval);
        }
      } catch (err) {
        console.error("Polling error:", err);
      }
    }, 3000);
    
    return () => clearInterval(pollInterval);
  }, [state]);
  
  useEffect(() => {
    if (state !== "qr_displayed" || timerSeconds <= 0) return;
    
    const timeout = setTimeout(() => {
      setTimerSeconds(prev => {
        if (prev <= 1) {
          setState("expired");
          return 0;
        }
        return prev - 1;
      });
    }, 1000);
    
    return () => clearInterval(timeout);
  }, [state, timerSeconds]);
  
  const handleConnectClick = async () => {
    setState("loading");
    setError(null);
    
    try {
      const response = await fetch("/api/settings/evo-go/qr", { method: "POST" });
      if (!response.ok) throw new Error("Failed to get QR code");
      
      const { qr_code } = await response.json();
      setQrCode(qr_code);
      setState("qr_displayed");
      setTimerSeconds(300);
    } catch (err) {
      setError(err instanceof Error ? err.message : "Unknown error");
      setState("error");
    }
  };
  
  return (
    <div className="settings-container">
      <Card title="WhatsApp Connection">
        {state === "idle" || state === "error" ? (
          <>
            <Badge variant={connectionStatus === "connected" ? "success" : "default"}>
              {connectionStatus === "connected" ? "Conectado ✅" : "Desconectado"}
            </Badge>
            <Button onClick={handleConnectClick}>Conectar WhatsApp</Button>
            {error && <div className="error-message">{error}</div>}
          </>
        ) : state === "loading" ? (
          <div>Carregando...</div>
        ) : state === "qr_displayed" ? (
          <>
            <img src={qrCode!} alt="QR Code" />
            <div>Expira em: {Math.floor(timerSeconds / 60)}:{String(timerSeconds % 60).padStart(2, "0")}</div>
          </>
        ) : state === "expired" ? (
          <>
            <div className="text-error">QR Code expirado.</div>
            <Button onClick={handleConnectClick}>Gerar novo QR</Button>
          </>
        ) : state === "connected" ? (
          <>
            <Badge variant="success">Conectado ✅</Badge>
            <Button>Desconectar</Button>
          </>
        ) : null}
      </Card>
    </div>
  );
}
```

### Webhook Security Model: tenantId Query Parameter

**Design Decision:** Extract `tenantId` from webhook URL query parameter (`?tenantId=...`)

**Rationale:**
- Evo GO webhooks don't include authentication (only HMAC signature)
- Webhook URL is registered during pairing and includes tenant context
- Query parameter allows webhook handler to identify which tenant to update without JWT

**Security Considerations:**
1. **HMAC-SHA256 validation is mandatory** — Ensures webhook comes from Evo GO, not random client
2. **tenantId parameter is not sensitive** — Already public in webhook URL, not a secret
3. **Database integrity** — Webhook updates only `connection_status`, which is safe to update
4. **Timing attack prevention** — Use constant-time comparison for HMAC (reuse Story 3.1)
