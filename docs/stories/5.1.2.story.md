**DIV-5 (CRÍTICA para create flow):**
- **Problema**: Documentação não lista `id` na resposta `/instance/create`, mas código exige
- **Atual**: `if (!responseData.id) throw new EvoGoError(...)`
- **Ação**: Adicionar fallback: se `id` não vem na resposta, fazer lookup via `GET /instance/all`
- **Implementação**:
  ```typescript
  async function callEvoGoCreateInstance(...) {
    const json = await response.json();
    const responseData = json.data || {};
    
    console.log('[EvoGo] /instance/create raw response', {
      timestamp: new Date().toISOString(),
      hasId: 'id' in responseData,
      hasName: 'name' in responseData,
      fields: Object.keys(responseData),
    });
    
    let instanceId = responseData.id;
    
    // Fallback: if no id, lookup by name
    if (!instanceId && responseData.name) {
      console.log('[EvoGo] No id in create response, looking up by name...', {
        name: responseData.name,
      });
      const instances = await listEvoGoInstances(apiKey);
      const created = instances.find(i => i.name === responseData.name);
      instanceId = created?.id;
    }
    
    if (!instanceId) {
      throw new EvoGoError(
        "Could not determine instance ID from create response or lookup",
        "MALFORMED_RESPONSE",
        500,
      );
    }
    
    return {
      instance_id: instanceId,
      token: responseData.token,
      qr_code: responseData.qrcode || '',
    };
  }
  ```
- **Fonte**: F5.plan.005.md — DIV-5

**DIV-7 (MÉDIA — format mismatch):**
- **Problema**: Documentação inconsistente (alguns eventos UPPERCASE, alguns dot.notation)
- **Atual**: Código usa UPPERCASE: `['QRCODE', 'CONNECTION', 'MESSAGE']`
- **Ação**: Testar qual formato API aceita; adicionar log; usar o que funciona
- **Implementação**:
  ```typescript
  // Test both formats to determine correct one
  async function determineSubscribeFormat(instanceToken: string): Promise<string[]> {
    const formats = [
      ['MESSAGE', 'CONNECTION', 'QRCODE'], // UPPERCASE
      ['messages.upsert', 'connection.update', 'qrcode'], // dot.notation
    ];
    
    for (const subscribe of formats) {
      try {
        const response = await fetch(`${EVOGO_API_BASE}/instance/connect`, {
          method: "POST",
          headers: { "apikey": instanceToken, "Content-Type": "application/json" },
          body: JSON.stringify({ subscribe, webhookUrl: '' }),
        });
        
        console.log('[EvoGo] /instance/connect format test', {
          format: subscribe[0].includes('.') ? 'dot.notation' : 'UPPERCASE',
          status: response.status,
          success: response.ok,
        });
        
        if (response.ok) return subscribe;
      } catch (e) {
        console.error('[EvoGo] Subscribe format test failed', { format: subscribe[0], error: e });
      }
    }
    
    // Fallback: use UPPERCASE (empirically observed in practice)
    console.warn('[EvoGo] Could not determine subscribe format, using UPPERCASE');
    return ['MESSAGE', 'CONNECTION', 'QRCODE'];
  }
  
  // Usage:
  const subscribeFormat = await determineSubscribeFormat(instanceToken);
  ```
- **Fonte**: F5.plan.005.md — DIV-7

**DIV-8 (MÉDIA — destructive operation clarity):**
- **Problema**: Código chama `DELETE /instance/logout` quando usuário clica "Desconectar", mas documentação distingue:
  - `POST /instance/disconnect` — suave, mantém sessão
  - `DELETE /instance/logout` — destrutivo, remove sessão
- **Ação**: Documentar a intenção explicitamente. Oferecer AMBOS os endpoints.
- **Implementação**:
  ```typescript
  /**
   * Soft disconnect: pausa conexão, mantém sessão
   * User pode reconectar sem rescanning QR code
   */
  export async function callEvoGoDisconnect(instanceToken: string): Promise<void> {
    const response = await fetch(`${EVOGO_API_BASE}/instance/disconnect`, {
      method: "POST",
      headers: {
        "apikey": instanceToken,
        "Content-Type": "application/json",
      },
    });

    if (!response.ok) {
      throw new EvoGoError(
        `Failed to disconnect instance: ${response.statusText}`,
        "API_ERROR",
        response.status,
      );
    }
    
    console.log('[EvoGo] /instance/disconnect soft disconnect', {
      timestamp: new Date().toISOString(),
      status: response.status,
    });
  }

  /**
   * Destructive logout: completa logout, remove sessão, exige novo QR scan
   * Use for explicit user logout or account cleanup
   */
  export async function callEvoGoLogout(instanceToken: string): Promise<void> {
    const response = await fetch(`${EVOGO_API_BASE}/instance/logout/${instanceToken}`, {
      method: "DELETE",
      headers: {
        "apikey": instanceToken,
      },
    });

    if (!response.ok) {
      throw new EvoGoError(
        `Failed to logout instance: ${response.statusText}`,
        "API_ERROR",
        response.status,
      );
    }
    
    console.log('[EvoGo] /instance/logout destructive logout', {
      timestamp: new Date().toISOString(),
      status: response.status,
      note: 'Session removed, user must rescan QR code to reconnect',
    });
  }
  ```
  
  **Decision Point for Settings UI:**
  - Button text "Desconectar" (pause) → call `callEvoGoDisconnect()` (soft)
  - Button text "Fazer Logout" (destructive) → call `callEvoGoLogout()` (hard)
  - Document in settings page which operation happens
- **Fonte**: F5.plan.005.md — DIV-8

**DIV-11 (MÉDIA — token in list response):**
- **Problema**: `GET /instance/all` não retorna `token` conforme documentação, mas código espera
- **Ação**: Adicionar fallback: se lista não tiver token, fazer lookup via `GET /instance/info/{id}`
- **Implementação**:
  ```typescript
  export async function getOrCreateInstance(
    apiKey: string,
    tenantId: string,
  ): Promise<EvoGoInstance> {
    const instances = await listEvoGoInstances(apiKey);
    
    if (instances.length > 0) {
      const instance = instances[0];
      
      console.log('[EvoGo] /instance/all response received', {
        instanceCount: instances.length,
        firstInstanceId: instance.id,
        hasToken: 'token' in instance,
      });
      
      // Fallback: if token missing, lookup via /instance/info/{id}
      if (!instance.token) {
        console.log('[EvoGo] Token missing in list, looking up instance info...', {
          instanceId: instance.id,
        });
        
        const info = await getEvoGoInstanceInfo(apiKey, instance.id);
        instance.token = info.token;
        
        console.log('[EvoGo] Instance token populated from /instance/info', {
          instanceId: instance.id,
          hasToken: !!instance.token,
        });
      }
      
      return instance;
    }
    
    // If not exists, create new
    return await callEvoGoCreateInstance(apiKey, {
      name: `kanban-instance-${tenantId.substring(0, 8)}`,
      webhookUrl: '',
    });
  }

  async function getEvoGoInstanceInfo(apiKey: string, instanceId: string): Promise<any> {
    const response = await fetch(`${EVOGO_API_BASE}/instance/info/${instanceId}`, {
      headers: { "apikey": apiKey },
    });

    if (!response.ok) {
      throw new EvoGoError(
        `Failed to fetch instance info: ${response.statusText}`,
        "API_ERROR",
        response.status,
      );
    }

    const json = await response.json();
    console.log('[EvoGo] /instance/info response', {
      timestamp: new Date().toISOString(),
      instanceId,
      hasToken: 'token' in (json.data || {}),
    });
    
    return json.data;
  }
  ```
- **Fonte**: F5.plan.005.md — DIV-11
